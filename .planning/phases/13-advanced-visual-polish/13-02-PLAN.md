    ```typescript
    import React from 'react';
    import { Canvas, Circle, BlurMask } from '@shopify/react-native-skia';
    import { useDerivedValue } from 'react-native-reanimated';
    import type { SharedValue } from 'react-native-reanimated';
    import { JOYSTICK_SIZE, MAX_BLUR_SIGMA, GLOW_CANVAS_SIZE } from './constants';

    const CENTER = GLOW_CANVAS_SIZE / 2;
    const RING_RADIUS = JOYSTICK_SIZE / 2;

    interface JoystickGlowCanvasProps {
      dragIntensity: SharedValue<number>;
      pillarColor: string;
    }

    export function JoystickGlowCanvas({ dragIntensity, pillarColor }: JoystickGlowCanvasProps) {
      // Derived blur sigma: 0 at idle, MAX_BLUR_SIGMA at full drag
      const blurSigma = useDerivedValue(() => dragIntensity.value * MAX_BLUR_SIGMA);

      return (
        <Canvas
          style={{
            position: 'absolute',
            width: GLOW_CANVAS_SIZE,
            height: GLOW_CANVAS_SIZE,
            marginLeft: -MAX_BLUR_SIGMA,
            marginTop: -MAX_BLUR_SIGMA,
          }}
          pointerEvents="none"
        >
          <Circle cx={CENTER} cy={CENTER} r={RING_RADIUS} color={pillarColor}>
            <BlurMask blur={blurSigma} style="outer" respectCTM={false} />
          </Circle>
        </Canvas>
      );
    }
    ```

    CRITICAL implementation notes:
    - `pointerEvents="none"` on the Canvas — prevents touch interception (Pitfall 1)
    - Canvas sized to GLOW_CANVAS_SIZE (140px) with negative margin of MAX_BLUR_SIGMA (20px) — centers over 100px outerRing with room for blur spread (Pitfall 2)
    - `blurSigma` is a `useDerivedValue` — Skia Recorder reads `.value` each frame (Pitfall 3)
    - `style="outer"` = exterior-only halo — nothing renders inside the circle
