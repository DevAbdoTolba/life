The oval rect is derived from translateX/translateY SharedValues — the axis aligned with drag gets a larger radius, producing directional elongation.

    ```typescript
    import React from 'react';
    import { Canvas, Oval, BlurMask } from '@shopify/react-native-skia';
    import { useDerivedValue } from 'react-native-reanimated';
    import type { SharedValue } from 'react-native-reanimated';
    import { JOYSTICK_SIZE, MAX_BLUR_SIGMA, GLOW_CANVAS_SIZE, STRETCH_AMOUNT } from './constants';

    const CENTER = GLOW_CANVAS_SIZE / 2;
    const DIAMETER = JOYSTICK_SIZE;  // Base diameter (100)

    interface JoystickGlowCanvasProps {
      dragIntensity: SharedValue<number>;
      translateX: SharedValue<number>;
      translateY: SharedValue<number>;
      pillarColor: string;
    }

    export function JoystickGlowCanvas({
      dragIntensity,
      translateX,
      translateY,
      pillarColor,
    }: JoystickGlowCanvasProps) {
      // Derived blur sigma: 0 at idle, MAX_BLUR_SIGMA at full drag
      const blurSigma = useDerivedValue(() => dragIntensity.value * MAX_BLUR_SIGMA);

      // Directional stretch: Oval rect with width/height driven by drag direction (D-01)
      // At rest (translateX=0, translateY=0): perfect circle (width=height=DIAMETER)
      // During drag: axis aligned with drag direction gets +STRETCH_AMOUNT pixels,
      // producing water-drop elongation toward the drag direction.
      // The stretch scales with dragIntensity so it's subtle at low drag and full at max.
      const ovalRect = useDerivedValue(() => {
        'worklet';
        const intensity = dragIntensity.value;
        const tx = translateX.value;
        const ty = translateY.value;

        // Normalize drag vector to get direction weights (0-1 per axis)
        const mag = Math.sqrt(tx * tx + ty * ty);
        const dirX = mag > 0.001 ? Math.abs(tx) / mag : 0;
        const dirY = mag > 0.001 ? Math.abs(ty) / mag : 0;

        // Stretch each axis proportional to its direction weight and intensity
        const stretchX = STRETCH_AMOUNT * dirX * intensity;
        const stretchY = STRETCH_AMOUNT * dirY * intensity;

        const w = DIAMETER + stretchX;
        const h = DIAMETER + stretchY;
        const x = CENTER - w / 2;
        const y = CENTER - h / 2;

        return { x, y, width: w, height: h };
      });

      return (
        <Canvas
          style={{
            position: 'absolute',
            width: GLOW_CANVAS_SIZE,
            height: GLOW_CANVAS_SIZE,
            marginLeft: -MAX_BLUR_SIGMA,
            marginTop: -MAX_BLUR_SIGMA,
          }}
          pointerEvents="none"
        >
          <Oval rect={ovalRect} color={pillarColor}>
            <BlurMask blur={blurSigma} style="outer" respectCTM={false} />
          </Oval>
        </Canvas>
      );
    }
    ```

    CRITICAL implementation notes:
    - Uses `Oval` instead of `Circle` to support directional stretch per D-01
    - `ovalRect` is a `useDerivedValue` returning `{x, y, width, height}` — Skia Recorder reads `.value` each frame
    - At rest (dragIntensity=0), stretchX and stretchY are both 0, so the Oval is a perfect circle (DIAMETER x DIAMETER)
    - During drag, the axis aligned with the drag direction gets up to +STRETCH_AMOUNT (4px) of elongation at max intensity
    - `pointerEvents="none"` on the Canvas — prevents touch interception (Pitfall 1)
    - Canvas sized to GLOW_CANVAS_SIZE (140px) with negative margin of MAX_BLUR_SIGMA (20px) — centers over 100px outerRing with room for blur spread (Pitfall 2)
    - `style="outer"` = exterior-only halo — nothing renders inside the oval
